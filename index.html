MainActivity

Esta es la actividad principal de la aplicación, sirviendo como punto de entrada. Se encarga de inicializar la interfaz de usuario, incluyendo campos de texto para que el usuario ingrese su nombre y edad. Contiene lógica para procesar estos datos y enviarlos a MainActivity2 mediante un Intent. Además, proporciona botones para navegar directamente a MainActivity4 y MainActivity5. 

import android.content.Intent;
import android.os.Bundle;
import android.view.View;
import android.widget.EditText;

import androidx.activity.EdgeToEdge;
import androidx.appcompat.app.AppCompatActivity;
import androidx.core.graphics.Insets;
import androidx.core.view.ViewCompat;
import androidx.core.view.WindowInsetsCompat;

public class MainActivity extends AppCompatActivity {
    EditText nombre, edad;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        EdgeToEdge.enable(this);
        setContentView(R.layout.activity_main);
        ViewCompat.setOnApplyWindowInsetsListener(findViewById(R.id.main), (v, insets) -> {
            Insets systemBars = insets.getInsets(WindowInsetsCompat.Type.systemBars());
            v.setPadding(systemBars.left, systemBars.top, systemBars.right, systemBars.bottom);
            return insets;
        });
        nombre = findViewById(R.id.nombre);
        edad = findViewById(R.id.edad);
    }

    public void procesar(View view) {
        String nombree = nombre.getText().toString();
        String edadd = edad.getText().toString();
        Intent intent = new Intent(MainActivity.this, MainActivity2.class);
        intent.putExtra("nombree", nombree);
        intent.putExtra("edadd", edadd);
        startActivity(intent);
    }

    public void m4(View view) {
        Intent intent = new Intent(MainActivity.this, MainActivity4.class);
        startActivity(intent);
    }

    public void m5(View view) {
        Intent intent = new Intent(MainActivity.this, MainActivity5.class);
        startActivity(intent);
    }
}

MainActivity3

El propósito de MainActivity3 es tomar la edad recibida (originalmente de MainActivity, pasada a través de MainActivity2) y mostrar un mensaje personalizado basado en rangos de edad. Realiza una validación para asegurar que la edad fue recibida correctamente antes de intentar procesarla. Categoriza la edad en "niño", "adolescente" o "adulto" y actualiza un TextView con el mensaje correspondiente.

            return;

        String mensaje = "Tienes "+edadd+"\n";
        if(edadd > 0 && edadd <= 12){
            mensaje += "eres un niño";
        }

        if(edadd > 12 && edadd <= 18){
            mensaje += "eres un adolescente";
        }

        if(edadd > 18 && edadd <= 60){
            mensaje += "eres un adulto";
        }

        txtmensaje.setText(mensaje);
    } // Cierre de onCreate añadido
} // Cierre de clase añadido

MiAdapter

MiAdapter es una clase crucial para el funcionamiento del RecyclerView en MainActivity5. Extiende RecyclerView.Adapter y se encarga de vincular la lista de objetos Contacto con las vistas individuales de cada ítem. Implementa los métodos necesarios como onCreateViewHolder (para inflar el layout de cada ítem), onBindViewHolder (para poblar los datos en la vista del ítem) y getItemCount (para saber cuántos ítems hay). También define una clase interna MiViewHolder que mantiene las referencias a las vistas de cada ítem. 

import android.view.LayoutInflater;
import android.view.View;
import android.view.ViewGroup;
import android.widget.TextView;

import androidx.annotation.NonNull;
import androidx.recyclerview.widget.RecyclerView;

import java.util.List;


public class MiAdapter extends RecyclerView.Adapter<MiAdapter.MiViewHolder> {
    List<Contacto> misDatos;

    public MiAdapter(List<Contacto> misDatos) {
        this.misDatos = misDatos;
    }

    @NonNull
    @Override
    public MiAdapter.MiViewHolder onCreateViewHolder(@NonNull ViewGroup parent, int viewType) {
        View v = LayoutInflater.from(parent.getContext())
                .inflate(R.layout.mi_layout_rv, parent, false);
        return new MiViewHolder(v);
    }

    @Override
    public void onBindViewHolder(@NonNull MiAdapter.MiViewHolder holder, int position) {
        holder.tvNombres.setText(misDatos.get(position).getNombres());
        holder.tvMensaje.setText(misDatos.get(position).getMensaje());
    }

    @Override
    public int getItemCount() {
        return misDatos.size();
    }

    public static class MiViewHolder extends RecyclerView.ViewHolder {
        TextView tvNombres, tvMensaje;
        public MiViewHolder(@NonNull View itemView) {
            super(itemView);
            tvNombres = itemView.findViewById(R.id.tvNombres);
            tvMensaje = itemView.findViewById(R.id.tvMensaje);
        }
    }
}

Proyecto: Calculadora de Serie Fibonacci

Esta aplicación demuestra el uso de dos Activities para un flujo de datos sencillo. La primera Activity (`Fibonacci MainActivity`) solicita un número entero positivo `N`. Al presionar un botón, `N` se envía a la segunda Activity (`Fibonacci SecondActivity`) mediante un Intent. En la segunda Activity, se calcula y muestra la suma de los primeros `N` términos de la serie de Fibonacci (1, 1, 2, 3, 5, ...), utilizando un bucle (sin recursión). Se incluye validación para asegurar que `N` sea mayor que 0.

Fibonacci MainActivity (Java)

package com.example.fibonacci; // Puedes cambiar el nombre del paquete

import android.content.Intent;
import android.os.Bundle;
import android.view.View;
import android.widget.EditText;
import android.widget.Toast;

import androidx.activity.EdgeToEdge;
import androidx.appcompat.app.AppCompatActivity;
import androidx.core.graphics.Insets;
import androidx.core.view.ViewCompat;
import androidx.core.view.WindowInsetsCompat;

public class MainActivity extends AppCompatActivity {

    private EditText editTextN;
    public static final String EXTRA_N = "com.example.fibonacci.EXTRA_N"; // Clave para el Intent

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        EdgeToEdge.enable(this);
        setContentView(R.layout.activity_main);
        ViewCompat.setOnApplyWindowInsetsListener(findViewById(R.id.main), (v, insets) -> {
            Insets systemBars = insets.getInsets(WindowInsetsCompat.Type.systemBars());
            v.setPadding(systemBars.left, systemBars.top, systemBars.right, systemBars.bottom);
            return insets;
        });

        editTextN = findViewById(R.id.editTextN);
    }

    // Método llamado cuando se hace clic en el botón
    public void calculateFibonacci(View view) {
        String nString = editTextN.getText().toString();

        if (nString.isEmpty()) {
            Toast.makeText(this, "Por favor, ingresa un número.", Toast.LENGTH_SHORT).show();
            return;
        }

        int N;
        try {
            N = Integer.parseInt(nString);
        } catch (NumberFormatException e) {
            Toast.makeText(this, "Entrada inválida. Ingresa un número entero.", Toast.LENGTH_SHORT).show();
            return;
        }

        // Validación N > 0 antes de enviar
        if (N <= 0) {
            Toast.makeText(this, "El número N debe ser mayor que 0.", Toast.LENGTH_SHORT).show();
            return;
        }

        // Crear Intent para navegar a SecondActivity
        Intent intent = new Intent(MainActivity.this, SecondActivity.class);
        // Enviar el número N usando un extra en el Intent
        intent.putExtra(EXTRA_N, N);
        startActivity(intent);
    }
}

Fibonacci SecondActivity (Java)

package com.example.fibonacci; // Debe ser el mismo paquete que MainActivity

import android.content.Intent;
import android.os.Bundle;
import android.widget.TextView;
import android.widget.Toast;

import androidx.activity.EdgeToEdge;
import androidx.appcompat.app.AppCompatActivity;
import androidx.core.graphics.Insets;
import androidx.core.view.ViewCompat;
import androidx.core.view.WindowInsetsCompat;

public class SecondActivity extends AppCompatActivity {

    private TextView textViewResult;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        EdgeToEdge.enable(this);
        setContentView(R.layout.activity_second);
        ViewCompat.setOnApplyWindowInsetsListener(findViewById(R.id.main), (v, insets) -> {
            Insets systemBars = insets.getInsets(WindowInsetsCompat.Type.systemBars());
            v.setPadding(systemBars.left, systemBars.top, systemBars.right, systemBars.bottom);
            return insets;
        });

        textViewResult = findViewById(R.id.textViewResult);

        // Obtener el Intent que inició esta actividad
        Intent intent = getIntent();
        if (intent == null) {
            // Esto no debería ocurrir si MainActivity siempre envía un Intent
            Toast.makeText(this, "Error: No se recibió ningún Intent.", Toast.LENGTH_LONG).show();
            finish(); // Regresar a la actividad anterior
            return;
        }

        // Obtener el valor de N del Intent. El -1 es un valor por defecto si la clave no se encuentra.
        int N = intent.getIntExtra(MainActivity.EXTRA_N, -1);

        // Validación de N (si por alguna razón se pasó un valor no válido o el Intent era incorrecto)
        if (N <= 0) {
            Toast.makeText(this, "Error: El número N recibido no es válido. Regresando...", Toast.LENGTH_LONG).show();
            finish(); // Regresar a la MainActivity
            return;
        }

        // Calcular la suma de los primeros N términos de la serie Fibonacci
        // La serie se define como (1, 1, 2, 3, 5, ...)
        long sum = 0;
        if (N == 1) {
            sum = 1;
        } else if (N == 2) {
            sum = 2; // 1 + 1
        } else {
            long a = 1; // Primer término
            long b = 1; // Segundo término
            sum = a + b; // Suma inicial de los dos primeros términos (1+1=2)

            for (int i = 3; i <= N; i++) {
                long nextTerm = a + b;
                sum += nextTerm;
                a = b;
                b = nextTerm;
            }
        }

        // Mostrar el resultado
        String result = "La suma de los primeros " + N + " términos de Fibonacci es:\n" + sum;
        textViewResult.setText(result);
    }
}

Proyecto: Calculadora Aritmética Básica

Esta aplicación implementa las cuatro operaciones aritméticas básicas (suma, resta, multiplicación y división) entre dos números enteros. Utiliza dos Activities: la primera (`Calculadora MainActivity`) permite ingresar los números y seleccionar la operación mediante cuatro botones. La segunda Activity (`Calculadora ResultActivity`) muestra el resultado de la operación y un botón para regresar a la primera Activity.

Calculadora MainActivity (Java)

package com.example.basiccalculator; // Cambia el nombre de tu paquete si es diferente

public class MainActivity extends AppCompatActivity {

    private EditText editTextNum1;
    private EditText editTextNum2;

    // Claves para los extras del Intent
    public static final String EXTRA_NUM1 = "com.example.basiccalculator.NUM1";
    public static final String EXTRA_NUM2 = "com.example.basiccalculator.NUM2";
    public static final String EXTRA_OPERATION = "com.example.basiccalculator.OPERATION";

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        EdgeToEdge.enable(this);
        setContentView(R.layout.activity_main);
        ViewCompat.setOnApplyWindowInsetsListener(findViewById(R.id.main), (v, insets) -> {
            Insets systemBars = insets.getInsets(WindowInsetsCompat.Type.systemBars());
            v.setPadding(systemBars.left, systemBars.top, systemBars.right, systemBars.bottom);
            return insets;
        });

        editTextNum1 = findViewById(R.id.editTextNum1);
        editTextNum2 = findViewById(R.id.editTextNum2);
    }

    // Método que se llama al hacer clic en cualquiera de los botones de operación
    public void performOperation(View view) {
        String sNum1 = editTextNum1.getText().toString();
        String sNum2 = editTextNum2.getText().toString();

        if (sNum1.isEmpty() || sNum2.isEmpty()) {
            Toast.makeText(this, "Por favor, ingresa ambos números.", Toast.LENGTH_SHORT).show();
            return;
        }

        try {
            int num1 = Integer.parseInt(sNum1);
            int num2 = Integer.parseInt(sNum2);

            String operation = "";
            int viewId = view.getId();

            if (viewId == R.id.buttonAdd) {
                operation = "SUM";
            } else if (viewId == R.id.buttonSubtract) {
                operation = "SUBTRACT";
            } else if (viewId == R.id.buttonMultiply) {
                operation = "MULTIPLY";
            } else if (viewId == R.id.buttonDivide) {
                operation = "DIVIDE";
            }

            Intent intent = new Intent(MainActivity.this, ResultActivity.class);
            intent.putExtra(EXTRA_NUM1, num1);
            intent.putExtra(EXTRA_NUM2, num2);
            intent.putExtra(EXTRA_OPERATION, operation);
            startActivity(intent);

        } catch (NumberFormatException e) {
            Toast.makeText(this, "Entrada inválida. Ingresa números enteros.", Toast.LENGTH_SHORT).show();
        }
    }
}

Calculadora ResultActivity (Java)

package com.example.basiccalculator; // Cambia el nombre de tu paquete si es diferente

import android.content.Intent;
import android.os.Bundle;
import android.view.View;
import android.widget.TextView;
import android.widget.Toast;

import androidx.activity.EdgeToEdge;
import androidx.appcompat.app.AppCompatActivity;
import androidx.core.graphics.Insets;
import androidx.core.view.ViewCompat;
import androidx.core.view.WindowInsetsCompat;

public class ResultActivity extends AppCompatActivity {

    private TextView textViewResult;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        EdgeToEdge.enable(this);
        setContentView(R.layout.activity_result);
        ViewCompat.setOnApplyWindowInsetsListener(findViewById(R.id.main), (v, insets) -> {
            Insets systemBars = insets.getInsets(WindowInsetsCompat.Type.systemBars());
            v.setPadding(systemBars.left, systemBars.top, systemBars.right, systemBars.bottom);
            return insets;
        });

        textViewResult = findViewById(R.id.textViewResult);

        Intent intent = getIntent();
        if (intent != null) {
            int num1 = intent.getIntExtra(MainActivity.EXTRA_NUM1, 0);
            int num2 = intent.getIntExtra(MainActivity.EXTRA_NUM2, 0);
            String operation = intent.getStringExtra(MainActivity.EXTRA_OPERATION);

            double result = 0; // Usamos double para la división
            String resultText = "";

            switch (operation) {
                case "SUM":
                    result = num1 + num2;
                    resultText = num1 + " + " + num2 + " = " + String.format("%.0f", result);
                    break;
                case "SUBTRACT":
                    result = num1 - num2;
                    resultText = num1 + " - " + num2 + " = " + String.format("%.0f", result);
                    break;
                case "MULTIPLY":
                    result = num1 * num2;
                    resultText = num1 + " * " + num2 + " = " + String.format("%.0f", result);
                    break;
                case "DIVIDE":
                    if (num2 == 0) {
                        resultText = "Error: División por cero";
                        Toast.makeText(this, resultText, Toast.LENGTH_LONG).show();
                        // Opcional: regresar a la actividad anterior automáticamente en caso de error
                        // finish();
                        // return;
                    } else {
                        result = (double) num1 / num2; // Casteo a double para resultado con decimales
                        resultText = num1 + " / " + num2 + " = " + String.format("%.2f", result); // Formatear a 2 decimales
                    }
                    break;
                default:
                    resultText = "Operación no válida.";
                    break;
            }
            textViewResult.setText(resultText);
        } else {
            Toast.makeText(this, "Error: No se recibieron datos de la operación.", Toast.LENGTH_SHORT).show();
            finish(); // Regresar a la MainActivity
        }
    }

    // Método para el botón "Cerrar"
    public void closeActivity(View view) {
        finish(); // Cierra esta actividad y regresa a la anterior (MainActivity)
    }
}
