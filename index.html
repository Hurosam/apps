MainActivity

Esta es la actividad principal de la aplicación, sirviendo como punto de entrada. Se encarga de inicializar la interfaz de usuario, incluyendo campos de texto para que el usuario ingrese su nombre y edad. Contiene lógica para procesar estos datos y enviarlos a MainActivity2 mediante un Intent. Además, proporciona botones para navegar directamente a MainActivity4 y MainActivity5. 

import android.content.Intent;
import android.os.Bundle;
import android.view.View;
import android.widget.EditText;

import androidx.activity.EdgeToEdge;
import androidx.appcompat.app.AppCompatActivity;
import androidx.core.graphics.Insets;
import androidx.core.view.ViewCompat;
import androidx.core.view.WindowInsetsCompat;

public class MainActivity extends AppCompatActivity {
    EditText nombre, edad;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        EdgeToEdge.enable(this);
        setContentView(R.layout.activity_main);
        ViewCompat.setOnApplyWindowInsetsListener(findViewById(R.id.main), (v, insets) -> {
            Insets systemBars = insets.getInsets(WindowInsetsCompat.Type.systemBars());
            v.setPadding(systemBars.left, systemBars.top, systemBars.right, systemBars.bottom);
            return insets;
        });
        nombre = findViewById(R.id.nombre);
        edad = findViewById(R.id.edad);
    }

    public void procesar(View view) {
        String nombree = nombre.getText().toString();
        String edadd = edad.getText().toString();
        Intent intent = new Intent(MainActivity.this, MainActivity2.class);
        intent.putExtra("nombree", nombree);
        intent.putExtra("edadd", edadd);
        startActivity(intent);
    }

    public void m4(View view) {
        Intent intent = new Intent(MainActivity.this, MainActivity4.class);
        startActivity(intent);
    }

    public void m5(View view) {
        Intent intent = new Intent(MainActivity.this, MainActivity5.class);
        startActivity(intent);
    }
}

MainActivity2

MainActivity2 está diseñada para recibir y mostrar los datos enviados desde MainActivity. Extrae el nombre y la edad del Intent entrante y los presenta en un TextView. También incluye funcionalidad para pasar la edad recibida a una tercera actividad, MainActivity3, demostrando el flujo de datos entre múltiples pantallas. 

import android.content.Intent;
import android.os.Bundle;
import android.view.View;
import android.widget.TextView;

import androidx.activity.EdgeToEdge;
import androidx.appcompat.app.AppCompatActivity;
import androidx.core.graphics.Insets;
import androidx.core.view.ViewCompat;
import androidx.core.view.WindowInsetsCompat;

public class MainActivity2 extends AppCompatActivity {

    TextView datos;
    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        EdgeToEdge.enable(this);
        setContentView(R.layout.activity_main2);
        ViewCompat.setOnApplyWindowInsetsListener(findViewById(R.id.main), (v, insets) -> {
            Insets systemBars = insets.getInsets(WindowInsetsCompat.Type.systemBars());
            v.setPadding(systemBars.left, systemBars.top, systemBars.right, systemBars.bottom);
            return insets;
        });

        datos = findViewById(R.id.datos);
        String nombree = getIntent().getStringExtra("nombree");
        String edadd = getIntent().getStringExtra("edadd");

        datos.setText("Nombre: "+nombree+"\n"+
                        "Edad:"+edadd);
    }

    public void siguiente(View view) {
        Intent intent = new Intent(MainActivity2.this,MainActivity3.class);
        String edadd = getIntent().getStringExtra("edadd");
        intent.putExtra("edadd", edadd);
        startActivity(intent);
    }
}

MainActivity3

El propósito de MainActivity3 es tomar la edad recibida (originalmente de MainActivity, pasada a través de MainActivity2) y mostrar un mensaje personalizado basado en rangos de edad. Realiza una validación para asegurar que la edad fue recibida correctamente antes de intentar procesarla. Categoriza la edad en "niño", "adolescente" o "adulto" y actualiza un TextView con el mensaje correspondiente. 

import android.content.Intent; // Añadido para getIntent()
import android.os.Bundle;
import android.widget.TextView;

import androidx.activity.EdgeToEdge;
import androidx.appcompat.app.AppCompatActivity;
import androidx.core.graphics.Insets;
import androidx.core.view.ViewCompat;
import androidx.core.view.WindowInsetsCompat;

public class MainActivity3 extends AppCompatActivity {

    TextView txtmensaje;
    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        EdgeToEdge.enable(this);
        setContentView(R.layout.activity_main3);
        ViewCompat.setOnApplyWindowInsetsListener(findViewById(R.id.main), (v, insets) -> {
            Insets systemBars = insets.getInsets(WindowInsetsCompat.Type.systemBars());
            v.setPadding(systemBars.left, systemBars.top, systemBars.right, systemBars.bottom);
            return insets;
        });
        txtmensaje = findViewById(R.id.mostrarMensaje);

        String edaddStr = getIntent().getStringExtra("edadd");
        if (edaddStr == null || edaddStr.isEmpty()) {
            txtmensaje.setText("Edad no recibida");
            return;
        }

        int edadd = Integer.parseInt(edaddStr);
        String mensaje = "Tienes "+edadd+"\n";
        if(edadd > 0 && edadd <= 12){
            mensaje += "eres un niño";
        }

        if(edadd > 12 && edadd <= 18){
            mensaje += "eres un adolescente";
        }

        if(edadd > 18 && edadd <= 60){
            mensaje += "eres un adulto";
        }

        txtmensaje.setText(mensaje);
    } // Cierre de onCreate añadido
} // Cierre de clase añadido

MainActivity5

Esta actividad se centra en mostrar una lista de datos utilizando un RecyclerView. Inicializa el RecyclerView, establece un LinearLayoutManager para la disposición de los ítems, y crea una lista de objetos Contacto de ejemplo. Luego, instancia y asigna un adaptador personalizado (MiAdapter) al RecyclerView para poblarlo con los datos. 

import android.os.Bundle;

import androidx.activity.EdgeToEdge;
import androidx.appcompat.app.AppCompatActivity;
import androidx.core.graphics.Insets;
import androidx.core.view.ViewCompat;
import androidx.core.view.WindowInsetsCompat;
import androidx.recyclerview.widget.LinearLayoutManager;
import androidx.recyclerview.widget.RecyclerView;

import java.util.ArrayList;
import java.util.List;

public class MainActivity5 extends AppCompatActivity {

    RecyclerView miRV;
    RecyclerView.LayoutManager layoutManager;
    RecyclerView.Adapter miAdaptador;
    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        EdgeToEdge.enable(this);
        setContentView(R.layout.activity_main5);
        ViewCompat.setOnApplyWindowInsetsListener(findViewById(R.id.main), (v, insets) -> {
            Insets systemBars = insets.getInsets(WindowInsetsCompat.Type.systemBars());
            v.setPadding(systemBars.left, systemBars.top, systemBars.right, systemBars.bottom);
            return insets;
        });

        miRV = findViewById(R.id.miRV);

        layoutManager = new LinearLayoutManager(this);
        miRV.setLayoutManager(layoutManager);

        List<Contacto> datos = new ArrayList<>();
        datos.add(new Contacto("Diego Rosales", "Hola mundo"));
        datos.add(new Contacto("Samir Huerto", "Saludos"));
        miAdaptador = new MiAdapter(datos);
        miRV.setAdapter(miAdaptador);
    }
}

Contacto

La clase Contacto es un simple POJO (Plain Old Java Object) que sirve como modelo de datos. Representa la estructura de un contacto, conteniendo propiedades para los nombres y un mensaje. Incluye un constructor para inicializar estos campos y los métodos getter y setter estándar para acceder y modificar los datos del objeto Contacto. 

public class Contacto {

MiAdapter

MiAdapter es una clase crucial para el funcionamiento del RecyclerView en MainActivity5. Extiende RecyclerView.Adapter y se encarga de vincular la lista de objetos Contacto con las vistas individuales de cada ítem. Implementa los métodos necesarios como onCreateViewHolder (para inflar el layout de cada ítem), onBindViewHolder (para poblar los datos en la vista del ítem) y getItemCount (para saber cuántos ítems hay). También define una clase interna MiViewHolder que mantiene las referencias a las vistas de cada ítem. 

import android.view.LayoutInflater;
import android.view.View;
import android.view.ViewGroup;
import android.widget.TextView;

import androidx.annotation.NonNull;
import androidx.recyclerview.widget.RecyclerView;

import java.util.List;


public class MiAdapter extends RecyclerView.Adapter<MiAdapter.MiViewHolder> {
    List<Contacto> misDatos;

    public MiAdapter(List<Contacto> misDatos) {
        this.misDatos = misDatos;
    }

    @NonNull
    @Override
    public MiAdapter.MiViewHolder onCreateViewHolder(@NonNull ViewGroup parent, int viewType) {
        View v = LayoutInflater.from(parent.getContext())
                .inflate(R.layout.mi_layout_rv, parent, false);
        return new MiViewHolder(v);
    }

    @Override
    public void onBindViewHolder(@NonNull MiAdapter.MiViewHolder holder, int position) {
        holder.tvNombres.setText(misDatos.get(position).getNombres());
        holder.tvMensaje.setText(misDatos.get(position).getMensaje());
    }

    @Override
    public int getItemCount() {
        return misDatos.size();
    }

    public static class MiViewHolder extends RecyclerView.ViewHolder {
        TextView tvNombres, tvMensaje;
        public MiViewHolder(@NonNull View itemView) {
            super(itemView);
            tvNombres = itemView.findViewById(R.id.tvNombres);
            tvMensaje = itemView.findViewById(R.id.tvMensaje);
        }
    }
}

Layout (Inclusión de Toolbar)

Este fragmento de código XML demuestra cómo incluir un layout de Toolbar separado dentro de otro archivo de layout. Utilizando la etiqueta , se puede reutilizar un diseño de Toolbar común en múltiples actividades. El atributo android:id permite referenciar esta Toolbar incluida desde el código Java, mientras que layout="@layout/mi_toolbar" especifica el archivo de layout que contiene la definición de la Toolbar. 

<include
    android:id="@+id/mi_toolbar"
    layout="@layout/mi_toolbar"/>

Menu (mi_menu.xml)

Este es un archivo XML que define la estructura de un menú de opciones para ser usado en una ActionBar o Toolbar. Cada etiqueta representa una opción del menú, con atributos como android:id para identificación, android:title para el texto visible, y opcionalmente android:icon para un ícono. El atributo app:showAsAction controla cómo y cuándo el ítem del menú se muestra en la barra de acciones (por ejemplo, "always", "ifRoom", "never"). 

<menu xmlns:android="http://schemas.android.com/apk/res/android"
    xmlns:app="http://schemas.android.com/apk/res-auto">
    <item android:id="@+id/op1"
        android:title="Opcion 1"
        android:icon="@drawable/search_svgrepo_com"
        app:showAsAction="always"/>
    <item android:id="@+id/op2"
        android:title="Opcion 2"
        app:showAsAction="never"/>
    <item android:id="@+id/op3"
        android:title="Opcion 3"
        android:icon="@drawable/three_dots_vertical_svgrepo_com"
        app:showAsAction="always"/>
