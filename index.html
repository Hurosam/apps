MainActivity

Recopila datos del usuario y navega a otras Activities.

import android.content.Intent;
import android.os.Bundle;
import android.view.View;
import android.widget.EditText;

import androidx.activity.EdgeToEdge;
import androidx.appcompat.app.AppCompatActivity;
import androidx.core.graphics.Insets;
import androidx.core.view.ViewCompat;
import androidx.core.view.WindowInsetsCompat;

public class MainActivity extends AppCompatActivity {
    EditText nombre, edad;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        EdgeToEdge.enable(this);
        setContentView(R.layout.activity_main);
        ViewCompat.setOnApplyWindowInsetsListener(findViewById(R.id.main), (v, insets) -> {
            Insets systemBars = insets.getInsets(WindowInsetsCompat.Type.systemBars());
            v.setPadding(systemBars.left, systemBars.top, systemBars.right, systemBars.bottom);
            return insets;
        });
        nombre = findViewById(R.id.nombre);
        edad = findViewById(R.id.edad);
    }

    public void procesar(View view) {
        String nombree = nombre.getText().toString();
        String edadd = edad.getText().toString();
        Intent intent = new Intent(MainActivity.this, MainActivity2.class);
        intent.putExtra("nombree", nombree);
        intent.putExtra("edadd", edadd);
        startActivity(intent);
    }

    public void m4(View view) {
        Intent intent = new Intent(MainActivity.this, MainActivity4.class);
        startActivity(intent);
    }

    public void m5(View view) {
        Intent intent = new Intent(MainActivity.this, MainActivity5.class);
        startActivity(intent);
    }
}

MainActivity3

Muestra un mensaje según la edad recibida.

import android.content.Intent; // Añadido para getIntent()
import android.os.Bundle;
import android.widget.TextView;

import androidx.activity.EdgeToEdge;
import androidx.appcompat.app.AppCompatActivity;
import androidx.core.graphics.Insets;
import androidx.core.view.ViewCompat;
import androidx.core.view.WindowInsetsCompat;

public class MainActivity3 extends AppCompatActivity {

    TextView txtmensaje;
    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        EdgeToEdge.enable(this);
        setContentView(R.layout.activity_main3);
        ViewCompat.setOnApplyWindowInsetsListener(findViewById(R.id.main), (v, insets) -> {
            Insets systemBars = insets.getInsets(WindowInsetsCompat.Type.systemBars());
            v.setPadding(systemBars.left, systemBars.top, systemBars.right, systemBars.bottom);
            return insets;
        });
        txtmensaje = findViewById(R.id.mostrarMensaje);

        String edaddStr = getIntent().getStringExtra("edadd");
        if (edaddStr == null || edaddStr.isEmpty()) {
            txtmensaje.setText("Edad no recibida");
            return;
        }

        int edadd = Integer.parseInt(edaddStr);
        String mensaje = "Tienes "+edadd+"\n";
        if(edadd > 0 && edadd <= 12){
            mensaje += "eres un niño";
        }

        if(edadd > 12 && edadd <= 18){
            mensaje += "eres un adolescente";
        }

        if(edadd > 18 && edadd <= 60){
            mensaje += "eres un adulto";
        }

        txtmensaje.setText(mensaje);
    } // Cierre de onCreate añadido
} // Cierre de clase añadido

public class Contacto {
    private String nombres;
    private String mensaje;

    public Contacto(String nombres, String mensaje) {
        this.nombres = nombres;
        this.mensaje = mensaje;
    }

    public String getNombres() {
        return nombres;
    }

    public void setNombres(String nombres) {
        this.nombres = nombres;
    }

    public String getMensaje() {
        return mensaje;
    }

    public void setMensaje(String mensaje) {
        this.mensaje = mensaje;
    }
}

MiAdapter

Adaptador para el RecyclerView que muestra los datos de Contacto.

import android.view.LayoutInflater;
import android.view.View;
import android.view.ViewGroup;
import android.widget.TextView;

import androidx.annotation.NonNull;
import androidx.recyclerview.widget.RecyclerView;

import java.util.List;


public class MiAdapter extends RecyclerView.Adapter<MiAdapter.MiViewHolder> {
    List<Contacto> misDatos;

    public MiAdapter(List<Contacto> misDatos) {
        this.misDatos = misDatos;
    }

    @NonNull
    @Override
    public MiAdapter.MiViewHolder onCreateViewHolder(@NonNull ViewGroup parent, int viewType) {
        View v = LayoutInflater.from(parent.getContext())
                .inflate(R.layout.mi_layout_rv, parent, false);
        return new MiViewHolder(v);
    }

    @Override
    public void onBindViewHolder(@NonNull MiAdapter.MiViewHolder holder, int position) {
        holder.tvNombres.setText(misDatos.get(position).getNombres());
        holder.tvMensaje.setText(misDatos.get(position).getMensaje());
    }

    @Override
    public int getItemCount() {
        return misDatos.size();
    }

    public static class MiViewHolder extends RecyclerView.ViewHolder {
        TextView tvNombres, tvMensaje;
        public MiViewHolder(@NonNull View itemView) {
            super(itemView);
            tvNombres = itemView.findViewById(R.id.tvNombres);
            tvMensaje = itemView.findViewById(R.id.tvMensaje);
        }
    }
}
